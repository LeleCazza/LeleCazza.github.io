SICUREZZA INFORMATICA

  1. DIVIDETEVI IN N GRUPPI DI MASSIMO 3 PERSONE
  2. SCEGLIETE UNA CHIAVE SEGRETA
  3. SCEGLIETE UNA FRASE SEGRETA
  4. CRIPTATE LA FRASE SEGRETA SCELTA UTILIZZANDO LA CHIAVE SEGRETA SCELTA
  5. SCRIVETE SU N FOGLIETTI LA FRASE CRIPTATA OTTENUTA
  6. CONSEGNATE UN FOGLIETTO CON LA VOSTRA FRASE CRIPTATA AD OGNI GRUPPO
  7. DECIFRATE LE FRASI CRIPTATE DEGLI ALTRI GRUPPI
  8. OGNI VOLTA CHE RIUSCITE A DECRIPTARE UNA FRASE VI AGGIUDICATE 1 PUNTO
  9. QUANDO TUTTE LE FRASI SARANNO STATE DECRIPTATE VINCE IL GRUPPO CON PIÚ PUNTI


Esempio codice di forza bruta (compilatore online): 

                    using System;
                    namespace cifrario_cesare
                    {
                        class Program
                        {
                            static void Main(string[] args)
                            {
                                char[] frase;
                                int chiave = 1;
                                string risposta;
                                bool trovato = true;
                                Console.WriteLine("INSERISCI LA FRASE DA DECIFRARE: ");
                                frase = Console.ReadLine().ToLower().ToCharArray();
                                while (trovato)
                                {
                                    int i = 0;
                                    foreach (char c in frase)
                                    {
                                        if(((int)c) >= (int)'a' && ((int)c) <= (int)'z')
                                        {
                                            int x = (int)c + 1;
                                            if (x >= 123)
                                                x -= 26;
                                            frase[i] = (char)(x);
                                        }
                                        i++;
                                    }
                                    Console.WriteLine("UTILIZZO CHIAVE = " + chiave);
                                    Console.ForegroundColor = ConsoleColor.Red;
                                    Console.WriteLine(frase);
                                    Console.ForegroundColor = ConsoleColor.White;
                                    Console.WriteLine("LA DECODIFICA E' CORRETTA?");
                                    risposta = Console.ReadLine();
                                    if (risposta.Contains("s"))
                                    {
                                        trovato = false;
                                        Console.ForegroundColor = ConsoleColor.Green;
                                        Console.WriteLine(frase);
                                        Console.ForegroundColor = ConsoleColor.White;
                                    }
                                    else
                                        chiave++;
                                }
                            }
                        }
                    }
